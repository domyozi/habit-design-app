### 1. KGI 設定エンドポイント 🔵

**信頼性**: 🔵 *api-endpoints.md PATCH /goals/{goal_id}/kgi より*

**実装ファイル**: `backend/app/api/routes/goals.py`

```python
@router.patch("/{goal_id}/kgi", response_model=GoalWithKgiResponse)
async def set_goal_as_kgi(
    goal_id: str,
    request: SetKgiRequest,
    user_id: str = Depends(get_current_user_id),
    supabase: Client = Depends(get_supabase_client),
):
    """既存 Goal を KGI として設定する。target_date は必須。"""
    # 自分の Goal かを確認
    existing = supabase.table("goals").select("*").eq("id", goal_id).eq("user_id", user_id).single().execute()
    if not existing.data:
        raise HTTPException(status_code=404, detail="Goal not found")

    # KGI 属性を更新
    update_data = {
        "target_value": request.target_value,
        "unit": request.unit,
        "target_date": str(request.target_date),
        "metric_type": request.metric_type,
        "current_value": request.current_value,
    }
    result = supabase.table("goals").update(update_data).eq("id", goal_id).eq("user_id", user_id).execute()
    
    return build_goal_with_kgi_response(result.data[0])


def build_goal_with_kgi_response(goal_data: dict) -> GoalWithKgiResponse:
    """Goal データから KGI 計算フィールドを付与して GoalWithKgiResponse を構築する"""
    from datetime import date as date_type
    import math
    
    today = date_type.today()
    target_date = goal_data.get("target_date")
    target_value = goal_data.get("target_value")
    current_value = goal_data.get("current_value")
    
    # is_kgi: target_date が設定されている場合
    is_kgi = target_date is not None
    days_remaining = None
    is_expired = False
    achievement_rate = None
    
    if is_kgi:
        td = date_type.fromisoformat(target_date) if isinstance(target_date, str) else target_date
        days_remaining = (td - today).days
        is_expired = days_remaining < 0
    
    # 達成率の計算 (REQ-KGI-006)
    metric_type = goal_data.get("metric_type")
    if metric_type == "binary":
        achievement_rate = 100.0 if current_value == 1.0 else 0.0
    elif target_value and target_value != 0 and current_value is not None:
        if metric_type == "percentage":
            # 進捗率 = current_value / target_value * 100
            achievement_rate = min(100.0, (current_value / target_value) * 100)
        else:
            # numeric 型: 初期値からの変化率（簡易版）
            # target_value に向けてどれだけ近づいたかの割合
            # この計算は初期値が必要なため、シンプルに current/target で算出
            achievement_rate = min(100.0, (current_value / target_value) * 100)
    
    return GoalWithKgiResponse(
        **goal_data,
        is_kgi=is_kgi,
        days_remaining=days_remaining,
        is_expired=is_expired,
        achievement_rate=achievement_rate,
    )
```

### 2. KGI 現在値更新エンドポイント 🔵

**信頼性**: 🔵 *REQ-KGI-005 より*

```python
@router.patch("/{goal_id}/kgi/current-value", response_model=GoalWithKgiResponse)
async def update_kgi_current_value(
    goal_id: str,
    request: UpdateKgiCurrentValueRequest,
    user_id: str = Depends(get_current_user_id),
    supabase: Client = Depends(get_supabase_client),
):
    """KGI の現在値を更新する"""
    result = supabase.table("goals").update(
        {"current_value": request.current_value}
    ).eq("id", goal_id).eq("user_id", user_id).execute()
    
    if not result.data:
        raise HTTPException(status_code=404, detail="Goal not found or not a KGI")
    
    return build_goal_with_kgi_response(result.data[0])
```

def test_set_goal_as_kgi_success(client, auth_headers, existing_goal_id):

        f"/goals/{existing_goal_id}/kgi",
