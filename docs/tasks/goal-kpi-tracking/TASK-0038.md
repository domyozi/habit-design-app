### 2. KpiChart コンポーネント 🔵

**信頼性**: 🔵 *REQ-LOG-005, REQ-DASH-003 より*

**実装ファイル**: `frontend-v2/src/components/KpiChart.tsx`

```tsx
import { useEffect, useState } from 'react';
import {
  ResponsiveContainer,
  LineChart,
  Line,
  BarChart,
  Bar,
  XAxis,
  YAxis,
  Tooltip,
  CartesianGrid,
  ReferenceLine,
} from 'recharts';
import { fetchKpiLogs } from '../lib/api';
import type { Kpi, KpiChartDataPoint, KpiChartResponse } from '../types';

type Granularity = 'daily' | 'weekly' | 'monthly';

interface KpiChartProps {
  kpi: Kpi;
}

const GRANULARITY_RANGE: Record<Granularity, string> = {
  daily: '30d',
  weekly: '12w',
  monthly: '6m',
};

const GRANULARITY_LABELS: Record<Granularity, string> = {
  daily: '日次',
  weekly: '週次',
  monthly: '月次',
};

export function KpiChart({ kpi }: KpiChartProps) {
  const [granularity, setGranularity] = useState<Granularity>('daily');
  const [chartData, setChartData] = useState<KpiChartResponse | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    setLoading(true);
    setError(null);
    fetchKpiLogs(kpi.id, granularity, GRANULARITY_RANGE[granularity])
      .then((res) => setChartData(res.data))
      .catch(() => setError('データを取得できませんでした'))
      .finally(() => setLoading(false));
  }, [kpi.id, granularity]);

  const handleGranularityChange = (g: Granularity) => {
    setGranularity(g);
  };

  const dataPoints: KpiChartDataPoint[] = chartData?.data_points ?? [];
  const summary = chartData?.summary;

  return (
    <div data-testid="kpi-chart" className="space-y-4">
      {/* 粒度タブ */}
      <div data-testid="kpi-chart-tabs" role="tablist" className="flex gap-2">
        {(['daily', 'weekly', 'monthly'] as Granularity[]).map((g) => (
          <button
            key={g}
            role="tab"
            aria-selected={granularity === g}
            data-testid={`kpi-chart-tab-${g}`}
            onClick={() => handleGranularityChange(g)}
            className={`px-3 py-1 rounded ${
              granularity === g ? 'bg-blue-500 text-white' : 'bg-gray-100'
            }`}
          >
            {GRANULARITY_LABELS[g]}
          </button>
        ))}
      </div>

      {/* ローディング */}
      {loading && (
        <div data-testid="kpi-chart-loading" className="h-48 animate-pulse bg-gray-200 rounded" />
      )}

      {/* エラー */}
      {!loading && error && (
        <p data-testid="kpi-chart-error" className="text-red-500">
          {error}
        </p>
      )}

      {/* データなし */}
      {!loading && !error && dataPoints.length === 0 && (
        <p data-testid="kpi-chart-empty">記録がありません</p>
      )}

      {/* グラフ本体 */}
      {!loading && !error && dataPoints.length > 0 && (
        <div data-testid="kpi-chart-graph">
          <ResponsiveContainer width="100%" height={240}>
            {granularity === 'daily' ? (
              <LineChart data={dataPoints}>
                <CartesianGrid strokeDasharray="3 3" />
                <XAxis dataKey="date" tick={{ fontSize: 11 }} />
                <YAxis />
                <Tooltip />
                <Line
                  type="monotone"
                  dataKey="value"
                  stroke="#3b82f6"
                  connectNulls={false}
                  dot={false}
                />
                {kpi.target_value !== undefined && (
                  <ReferenceLine
                    y={kpi.target_value}
                    stroke="#ef4444"
                    strokeDasharray="4 4"
                    label={`目標: ${kpi.target_value}${kpi.unit ? ` ${kpi.unit}` : ''}`}
                    data-testid="kpi-chart-reference-line"
                  />
                )}
              </LineChart>
            ) : (
              <BarChart data={dataPoints}>
                <CartesianGrid strokeDasharray="3 3" />
                <XAxis dataKey="date" tick={{ fontSize: 11 }} />
                <YAxis />
                <Tooltip />
                <Bar dataKey="value" fill="#3b82f6" />
                {kpi.target_value !== undefined && (
                  <ReferenceLine
                    y={kpi.target_value}
                    stroke="#ef4444"
                    strokeDasharray="4 4"
                    label={`目標: ${kpi.target_value}${kpi.unit ? ` ${kpi.unit}` : ''}`}
                    data-testid="kpi-chart-reference-line"
                  />
                )}
              </BarChart>
            )}
          </ResponsiveContainer>
        </div>
      )}

      {/* サマリー統計 */}
      {!loading && !error && summary && (
        <div data-testid="kpi-chart-summary" className="grid grid-cols-4 gap-2">
          <div data-testid="kpi-chart-summary-avg" className="bg-gray-50 rounded p-2 text-center">
            <p className="text-xs text-gray-500">平均</p>
            <p className="font-bold">{summary.average?.toFixed(1) ?? '—'}</p>
          </div>
          <div data-testid="kpi-chart-summary-max" className="bg-gray-50 rounded p-2 text-center">
            <p className="text-xs text-gray-500">最大</p>
            <p className="font-bold">{summary.max?.toFixed(1) ?? '—'}</p>
          </div>
          <div data-testid="kpi-chart-summary-min" className="bg-gray-50 rounded p-2 text-center">
            <p className="text-xs text-gray-500">最小</p>
            <p className="font-bold">{summary.min?.toFixed(1) ?? '—'}</p>
          </div>
          <div data-testid="kpi-chart-summary-latest" className="bg-gray-50 rounded p-2 text-center">
            <p className="text-xs text-gray-500">最新</p>
            <p className="font-bold">
              {dataPoints.length > 0
                ? (dataPoints[dataPoints.length - 1].value?.toFixed(1) ?? '—')
                : '—'}
            </p>
          </div>
        </div>
      )}
    </div>
  );
}
```

### 3. KpiChart の配置 🟡

**信頼性**: 🟡 *UI 設計から妥当な推測（詳細は TASK-0035/0036 と調整）*

KPI カードを展開したとき（またはKPI詳細画面）に `KpiChart` を表示する。既存の `KpiCard.tsx` に組み込むか、アコーディオン形式で展開する。

```tsx
// KpiCard.tsx（既存コンポーネントへの組み込みイメージ）
import { useState } from 'react';
import { KpiChart } from './KpiChart';
import type { Kpi } from '../types';

export function KpiCard({ kpi }: { kpi: Kpi }) {
  const [expanded, setExpanded] = useState(false);

  return (
    <div className="border rounded-lg p-4">
      <div
        className="flex justify-between items-center cursor-pointer"
        onClick={() => setExpanded((v) => !v)}
      >
        <h3>{kpi.title}</h3>
        <span>{expanded ? '▲' : '▼'}</span>
      </div>
      {expanded && <KpiChart kpi={kpi} />}
    </div>
  );
}
```

import { render, screen } from '@testing-library/react';

vi.mock('../lib/api');

it('週次タブをクリックするとweeklyで再取得される', async () => {

    success: true,

      granularity: 'weekly',

  render(<KpiChart kpi={mockKpi} />);

## 注意事項

- `recharts` は `ResponsiveContainer` で包んでレスポンシブ対応する（`width="100%"` 必須）
- `LineChart` の `connectNulls={false}` を明示的に指定して null データポイントをギャップとして表示する
- Recharts の SVG はテスト環境（jsdom）では完全にレンダリングされないため、グラフ要素のアサートには `data-testid` を用いる
- `ReferenceLine` は Recharts コンポーネントとして描画されるため、テストではテキストラベルかコンテナの存在でアサートする
- 月次タブ切り替え時は `fetchKpiLogs` が `'monthly'` と `'6m'` で再呼び出しされることを確認する
- `summary.completion_rate` は `binary` 型 KPI の場合のみ有効（他型では `undefined`）
