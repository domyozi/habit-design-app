### 1. 型定義追加 🔵

**信頼性**: 🔵 *interfaces.ts より*

KGI/KPI 機能に必要な全型定義を `frontend-v2/src/types/index.ts` に追記する。既存型との命名衝突がないよう注意し、既存の `Goal` 型を拡張する形で `GoalWithKgi` を定義する。

**実装ファイル**: `frontend-v2/src/types/index.ts`

```typescript
// ─── Enum-like string unions ───────────────────────────────────────────────

export type MetricType = 'numeric' | 'percentage' | 'binary';
export type TrackingFrequency = 'daily' | 'weekly' | 'monthly';
export type KpiInputMethod = 'manual' | 'voice' | 'auto';

// ─── KGI (Key Goal Indicator) ──────────────────────────────────────────────

/** Goal を KGI として設定した場合の拡張型 */
export interface GoalWithKgi {
  id: string;
  user_id: string;
  title: string;
  description?: string;
  is_kgi: boolean;
  // KGI 固有フィールド（is_kgi=true の場合のみ意味を持つ）
  target_value?: number;
  current_value?: number;
  unit?: string;
  target_date?: string; // ISO 8601 date string (YYYY-MM-DD)
  metric_type?: MetricType;
  // 算出フィールド（バックエンドが計算して返す）
  achievement_rate?: number; // 0〜100 (%)
  days_remaining?: number;   // 負の場合は超過日数
  is_expired?: boolean;
  created_at: string;
  updated_at: string;
}

export interface SetKgiRequest {
  target_value?: number;
  current_value?: number;
  unit?: string;
  target_date: string; // 必須
  metric_type: MetricType;
}

export interface UpdateKgiCurrentValueRequest {
  current_value: number;
}

// ─── KPI (Key Performance Indicator) ──────────────────────────────────────

export interface Kpi {
  id: string;
  user_id: string;
  goal_id: string;
  title: string;
  description?: string;
  metric_type: MetricType;
  target_value?: number;
  unit?: string;
  tracking_frequency: TrackingFrequency;
  display_order: number;
  is_active: boolean;
  created_at: string;
  updated_at: string;
  // オプション（レスポンスによって含まれる場合がある）
  today_log?: KpiLog;
  habit_ids?: string[];
}

export interface CreateKpiRequest {
  goal_id: string;
  title: string;
  description?: string;
  metric_type: MetricType;
  target_value?: number;
  unit?: string;
  tracking_frequency: TrackingFrequency;
  display_order?: number;
}

// ─── KPI Log ───────────────────────────────────────────────────────────────

export interface KpiLog {
  id: string;
  kpi_id: string;
  user_id: string;
  log_date: string; // YYYY-MM-DD
  value: number;
  input_method: KpiInputMethod;
  note?: string;
  created_at: string;
  updated_at: string;
}

export interface UpsertKpiLogRequest {
  log_date: string; // YYYY-MM-DD
  value: number;
  input_method?: KpiInputMethod;
  note?: string;
}

// ─── KPI Chart ─────────────────────────────────────────────────────────────

export interface KpiChartDataPoint {
  date: string;   // YYYY-MM-DD
  value: number | null;
}

export interface KpiChartResponse {
  kpi_id: string;
  granularity: 'daily' | 'weekly' | 'monthly';
  data_points: KpiChartDataPoint[];
  summary: {
    average?: number;
    max?: number;
    min?: number;
    total?: number;
    completion_rate?: number; // binary 型の場合
  };
}

// ─── KPI with today status ─────────────────────────────────────────────────

export interface KpiWithTodayStatus extends Kpi {
  today_completed: boolean;
  today_value: number | null;
  connected_habits: Array<{
    id: string;
    title: string;
  }>;
}

// ─── Voice / Weekly summary ────────────────────────────────────────────────

export interface VoiceKpiUpdateResponse {
  success: boolean;
  updated_kpis: Array<{
    kpi_id: string;
    kpi_title: string;
    value: number;
    log: KpiLog;
  }>;
  raw_transcript?: string;
}

export interface WeeklyKgiSummary {
  goal_id: string;
  goal_title: string;
  kgi: GoalWithKgi;
  kpis: Array<{
    kpi: Kpi;
    week_logs: KpiLog[];
    week_average?: number;
    week_completion_rate?: number;
  }>;
  overall_achievement_rate: number;
}
```

### 2. APIクライアント追加 🔵

**信頼性**: 🔵 *api-endpoints.md より*

`frontend-v2/src/lib/api.ts` に KGI/KPI エンドポイントに対応する API 関数を追加する。既存ファイルの fetch ラッパーパターン（Authorization ヘッダー付与、エラーハンドリング）に従う。

**実装ファイル**: `frontend-v2/src/lib/api.ts`

```typescript
// ─── KGI API ───────────────────────────────────────────────────────────────

/**
 * Goal を KGI として設定（または KGI 属性を更新）する
 * PATCH /goals/{goal_id}/kgi
 */
export async function patchGoalKgi(
  goalId: string,
  req: SetKgiRequest,
): Promise<{ success: boolean; data: GoalWithKgi }> {
  const res = await apiFetch(`/goals/${goalId}/kgi`, {
    method: 'PATCH',
    body: JSON.stringify(req),
  });
  return res.json();
}

/**
 * KGI の現在値のみを更新する（軽量エンドポイント）
 * PATCH /goals/{goal_id}/kgi/current-value
 */
export async function patchGoalKgiCurrentValue(
  goalId: string,
  currentValue: number,
): Promise<{ success: boolean; data: GoalWithKgi }> {
  const req: UpdateKgiCurrentValueRequest = { current_value: currentValue };
  const res = await apiFetch(`/goals/${goalId}/kgi/current-value`, {
    method: 'PATCH',
    body: JSON.stringify(req),
  });
  return res.json();
}

/**
 * KGI 情報付きで Goal 一覧を取得する
 * GET /goals?include_kgi=true
 */
export async function fetchGoalsWithKgi(): Promise<{
  success: boolean;
  data: GoalWithKgi[];
}> {
  const res = await apiFetch('/goals?include_kgi=true');
  return res.json();
}

// ─── KPI API ───────────────────────────────────────────────────────────────

/**
 * KPI を作成する
 * POST /kpis
 */
export async function createKpi(
  req: CreateKpiRequest,
): Promise<{ success: boolean; data: Kpi }> {
  const res = await apiFetch('/kpis', {
    method: 'POST',
    body: JSON.stringify(req),
  });
  return res.json();
}

/**
 * 指定 Goal に紐づく KPI 一覧を取得する
 * GET /goals/{goal_id}/kpis
 */
export async function fetchKpis(
  goalId: string,
): Promise<{ success: boolean; data: Kpi[] }> {
  const res = await apiFetch(`/goals/${goalId}/kpis`);
  return res.json();
}

/**
 * 今日の KPI 一覧（today_log 付き）を取得する
 * GET /kpis/today
 */
export async function fetchKpisToday(): Promise<{
  success: boolean;
  data: KpiWithTodayStatus[];
}> {
  const res = await apiFetch('/kpis/today');
  return res.json();
}

/**
 * KPI ログを登録または更新する（upsert）
 * PUT /kpis/{kpi_id}/logs
 */
export async function upsertKpiLog(
  kpiId: string,
  req: UpsertKpiLogRequest,
): Promise<{ success: boolean; data: KpiLog }> {
  const res = await apiFetch(`/kpis/${kpiId}/logs`, {
    method: 'PUT',
    body: JSON.stringify(req),
  });
  return res.json();
}

/**
 * KPI のグラフデータを取得する
 * GET /kpis/{kpi_id}/logs/chart?granularity=daily&range=30d
 */
export async function fetchKpiLogs(
  kpiId: string,
  granularity: 'daily' | 'weekly' | 'monthly',
  range?: string,
): Promise<{ success: boolean; data: KpiChartResponse }> {
  const params = new URLSearchParams({ granularity });
  if (range) params.set('range', range);
  const res = await apiFetch(`/kpis/${kpiId}/logs/chart?${params.toString()}`);
  return res.json();
}

/**
 * KPI に習慣を紐づける（全上書き）
 * PUT /kpis/{kpi_id}/habits
 */
export async function linkKpiHabits(
  kpiId: string,
  habitIds: string[],
): Promise<{ success: boolean; data: { kpi_id: string; habit_ids: string[] } }> {
  const res = await apiFetch(`/kpis/${kpiId}/habits`, {
    method: 'PUT',
    body: JSON.stringify({ habit_ids: habitIds }),
  });
  return res.json();
}

/**
 * KPI を削除する
 * DELETE /kpis/{kpi_id}
 */
export async function deleteKpi(
  kpiId: string,
): Promise<{ success: boolean }> {
  const res = await apiFetch(`/kpis/${kpiId}`, {
    method: 'DELETE',
  });
  return res.json();
}
```
