```python
from datetime import date, timedelta
from typing import Literal
from statistics import mean

@router.get("/{kpi_id}/logs", response_model=KpiChartResponse)
async def get_kpi_logs_chart(
    kpi_id: str,
    granularity: Literal["daily", "weekly", "monthly"] = "daily",
    range: str = Query("30d", regex="^(\\d+d|\\d+w|\\d+m)$"),
    user_id: str = Depends(get_current_user_id),
    supabase: Client = Depends(get_supabase_client),
):
    """KPI ログをグラフ用に集計して返す"""
    # KPI 取得（所有権確認）
    kpi = supabase.table("kpis").select("*").eq("id", kpi_id).eq("user_id", user_id).single().execute()
    if not kpi.data:
        raise HTTPException(status_code=404, detail="KPI not found")
    
    # range を日数に変換
    today = date.today()
    start_date = _parse_range_to_start_date(range, today)
    
    # ログ取得
    logs = supabase.table("kpi_logs").select("log_date, value").eq("kpi_id", kpi_id).gte("log_date", str(start_date)).order("log_date").execute()
    log_map = {l["log_date"]: l["value"] for l in logs.data}
    
    # 粒度別に集計
    data_points = _aggregate_by_granularity(log_map, start_date, today, granularity)
    
    # サマリー計算
    values = [dp.value for dp in data_points if dp.value is not None]
    summary = KpiChartSummary(
        avg=round(mean(values), 2) if values else None,
        max=max(values) if values else None,
        min=min(values) if values else None,
        latest_value=log_map.get(str(today)),
        target_value=kpi.data.get("target_value"),
    )
    
    return KpiChartResponse(
        kpi_id=kpi_id,
        granularity=granularity,
        data_points=data_points,
        summary=summary,
    )


def _parse_range_to_start_date(range_str: str, today: date) -> date:
    """'30d', '12w', '6m' などの range 文字列を開始日に変換"""
    if range_str.endswith("d"):
        return today - timedelta(days=int(range_str[:-1]))
    elif range_str.endswith("w"):
        return today - timedelta(weeks=int(range_str[:-1]))
    elif range_str.endswith("m"):
        # 月数での近似（30日/月）
        return today - timedelta(days=int(range_str[:-1]) * 30)
    return today - timedelta(days=30)


def _aggregate_by_granularity(log_map, start_date, today, granularity):
    """粒度別に集計してデータポイントを生成する"""

def test_get_daily_logs_with_null_for_no_record(client, auth_headers, kpi_id):
    # 一部の日だけログを作成
    client.put(f"/kpis/{kpi_id}/logs", json={"log_date": "2026-04-15", "value": 74.5}, headers=auth_headers)
    
    response = client.get(f"/kpis/{kpi_id}/logs?granularity=daily&range=7d", headers=auth_headers)
    assert response.status_code == 200
    data_points = response.json()["data"]["data_points"]
    
    # 記録がない日は null
    non_recorded_days = [dp for dp in data_points if dp["date"] != "2026-04-15"]
    assert all(dp["value"] is None for dp in non_recorded_days)
    
    # 記録がある日は値を返す
    recorded_day = next(dp for dp in data_points if dp["date"] == "2026-04-15")
    assert recorded_day["value"] == 74.5
```

def test_kpi_chart_summary(client, auth_headers, kpi_id):
    # 複数日のログを作成
    for d, v in [("2026-04-13", 75.0), ("2026-04-14", 74.5), ("2026-04-15", 74.0)]:
        client.put(f"/kpis/{kpi_id}/logs", json={"log_date": d, "value": v}, headers=auth_headers)
    
    response = client.get(f"/kpis/{kpi_id}/logs?granularity=daily&range=7d", headers=auth_headers)
    summary = response.json()["data"]["summary"]
    assert summary["max"] == 75.0
    assert summary["min"] == 74.0
    assert summary["latest_value"] == 74.0
