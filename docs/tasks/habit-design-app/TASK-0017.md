### 2. SSEストリーミングカスタムフック 🔵

**信頼性**: 🔵 *design-interview.md Q5（ストリーミング）・NFR-002 より*

**実装ファイル**: `frontend/src/hooks/useSSEStream.ts`

```typescript
export function useSSEStream(url: string, enabled: boolean) {
  const [chunks, setChunks] = useState<string[]>([]);
  const [isDone, setIsDone] = useState(false);
  const [suggestedGoals, setSuggestedGoals] = useState<Goal[]>([]);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!enabled) return;
    const eventSource = new EventSource(url, { withCredentials: true });
    eventSource.onmessage = (e) => {
      const data = JSON.parse(e.data);
      if (data.type === 'chunk') setChunks(prev => [...prev, data.content]);
      if (data.type === 'done') {
        setIsDone(true);
        setSuggestedGoals(data.suggested_goals ?? []);
        eventSource.close();
      }
      if (data.type === 'error') {
        setError(data.error);
        eventSource.close();
      }
    };
    return () => eventSource.close();
  }, [url, enabled]);

  return { chunks, isDone, suggestedGoals, error };
}
```
