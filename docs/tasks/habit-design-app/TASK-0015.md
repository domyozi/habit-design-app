### 2. 習慣ログ更新カスタムフック 🔵

**信頼性**: 🔵 *REQ-501/502 より*

**実装ファイル**: `frontend/src/hooks/useHabitLog.ts`

```typescript
export function useHabitLog() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ habitId, completed, date }: UpdateHabitLogRequest) =>
      apiPatch(`/habits/${habitId}/log`, { date, completed }),
    onMutate: async (variables) => {
      // 楽観的更新
      await queryClient.cancelQueries({ queryKey: ['habits'] });
      const previous = queryClient.getQueryData(['habits']);
      // ローカルキャッシュを更新
      return { previous };
    },
    onError: (_, __, context) => {
      // エラー時にロールバック
      queryClient.setQueryData(['habits'], context?.previous);
    },
    onSuccess: (data) => {
      // バッジ獲得確認
      if (data.badge_earned) {
        // BadgeEarnedModal表示
      }
    },
  });
}
```
